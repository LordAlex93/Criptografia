a) Probar que \(561\) es un número de Carmichael, sin usar el teorema de caracterización. Hacer lo mismo con \(1729\), \(10585\), \(75361\) y \(1024651\).

def carmichael(n):
    L = [a for a in range(0,n) if gcd(n,a)==1]
    Zn = IntegerModRing(n)
    t = True
    i = 0
    while t and i < len(L):
        if Zn(L[i])^(n-1)-1 != 0:
            t = False
        else:
            i += 1
    return t

2. Utilizar los tests de Fermat y Miller-Rabin (con \(10\) repeticiones) para encontrar un testigo de pseudoprimalidad o para concluir que el número es probablemente primo.

def testFermat(n,t):
    Zn = IntegerModRing(n)
    if n < 3 or is_even(n):
        raise Exception("Valor de n = "+str(n)+" no válido.")
    for i in range(0,t):
        a = randint(2,n-2)
        if gcd(a,n) != 1:
            return "Compuesto"
        r = Zn(a)^(n-1)
        if r != 1:
            return "Compuesto"
    return "Primo"

def testMillerRabin(n,t):
    Zn = IntegerModRing(n)
    if n < 3 or is_even(n):
        raise Exception("Valor de n = "+str(n)+" no válido.")
    k = factor(n-1)[0][1]
    m = (n-1)/(2**k)
    for j in range(0,t):
        a = randint(2,n-1)
        if gcd(a,n) != 1:
            return "Compuesto"
        b = Zn(a)^m
        if Zn(b) == 1:
            return "Primo"
        for i in range(0,k):
            if Zn(b) == -1:
                return "Primo"
            else:
                b = Zn(b)^2
    return "Compuesto"

3. Repetir el ejercicio anterior usando el criterio de Solovay-Strassen.

def testSolovayStrassen(n,t):
    Zn = IntegerModRing(n)
    if n < 3 or is_even(n):
        raise Exception("Valor de n = "+str(n)+" no válido.")
    for i in range(0,t):
        a = randint(2,n-2)
        r = Zn(a)^((n-1)/2)
        if r != 1 and r != n-1:
            return "Compuesto"
        s = jacobi_symbol(a,n)
        if r != Zn(s):
            return "Compuesto"
    return "Primo"
for k in [na,nb,nc,nd,ne]:
