Algoritmo p-1 de Pollard

def mcm(B):
    m = lcm(1,2)
    mm = 0
    if B == 1:
        return 1
    for k in range(3,B+1):
        mm = lcm(m,k)
        m = mm
    return m
def pollardp(n,B,a,A):
    m = mcm(B)
    x = (a**m-1) % n
    g = gcd(x,n)
    if g != 1:
        return g
    while a < A:
        a+=1
        x = (a**m-1) % n
        g = gcd(x,n)
        if g != 1:
            return g

def fermat(n):
    t = int(sqrt(n))+1
    s = t**2-n
    while sqrt(s) != int(sqrt(s)):
        s = t**2-n
        t +=1
    return (t-sqrt(s),t+sqrt(s))

def fermatext(n,k,binit):
    b = binit
    a = sqrt(k*n+b^2)
    while a != int(a):
        b += 1
        a = sqrt(k*n+b^2)
    m1 = gcd(k*n,a-b)
    m2 = gcd(k*n,a+b)
    return m1,m2
